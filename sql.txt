-- =========================================
-- Alumni Giving: example Postgres schema
-- (tables + constraints + indexes)
-- =========================================

-- For UUID generation (gen_random_uuid())
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- -------------------------
-- 1) profiles
-- -------------------------
-- Stores alumni-specific data keyed by auth user id.
-- In Supabase later: user_id should match auth.users.id (UUID).
CREATE TABLE IF NOT EXISTS profiles (
  user_id          uuid PRIMARY KEY,
  email            text UNIQUE,                  -- optional (auth provider is source of truth)
  full_name        text,
  graduation_year  integer NOT NULL,
  role             text NOT NULL DEFAULT 'user',
  created_at       timestamptz NOT NULL DEFAULT now(),
  updated_at       timestamptz NOT NULL DEFAULT now(),

  -- Keep DB constraints "timeless" (avoid now() in CHECK). Enforce business windows in app too.
  CONSTRAINT profiles_graduation_year_range
    CHECK (graduation_year BETWEEN 1900 AND 2100),

  CONSTRAINT profiles_role_allowed
    CHECK (role IN ('user', 'admin'))
);

CREATE INDEX IF NOT EXISTS idx_profiles_graduation_year
  ON profiles (graduation_year);

CREATE INDEX IF NOT EXISTS idx_profiles_role
  ON profiles (role);

-- -------------------------
-- 2) billing_customers
-- -------------------------
-- Maps a user to a Stripe Customer ID.
CREATE TABLE IF NOT EXISTS billing_customers (
  user_id            uuid PRIMARY KEY REFERENCES profiles(user_id) ON DELETE CASCADE,
  stripe_customer_id text NOT NULL UNIQUE,
  created_at         timestamptz NOT NULL DEFAULT now()
);

-- Helpful for webhook lookups
CREATE INDEX IF NOT EXISTS idx_billing_customers_stripe_customer_id
  ON billing_customers (stripe_customer_id);

-- -------------------------
-- 3) billing_subscriptions
-- -------------------------
-- Maps a user to a Stripe Subscription ID + cached state for UI/admin.
CREATE TABLE IF NOT EXISTS billing_subscriptions (
  user_id                 uuid PRIMARY KEY REFERENCES profiles(user_id) ON DELETE CASCADE,
  stripe_subscription_id   text NOT NULL UNIQUE,
  status                  text NOT NULL DEFAULT 'incomplete',
  current_quantity        integer,               -- cached from Stripe (debug/UX)
  current_period_end      timestamptz,           -- cached from Stripe (UX)
  cancel_at_period_end    boolean NOT NULL DEFAULT false,
  updated_at              timestamptz NOT NULL DEFAULT now(),

  CONSTRAINT billing_subscriptions_status_allowed
    CHECK (status IN (
      'incomplete',
      'incomplete_expired',
      'trialing',
      'active',
      'past_due',
      'canceled',
      'unpaid',
      'paused'
    )),

  CONSTRAINT billing_subscriptions_quantity_nonnegative
    CHECK (current_quantity IS NULL OR current_quantity >= 0)
);

CREATE INDEX IF NOT EXISTS idx_billing_subscriptions_status
  ON billing_subscriptions (status);

CREATE INDEX IF NOT EXISTS idx_billing_subscriptions_period_end
  ON billing_subscriptions (current_period_end);

-- -------------------------
-- 4) stripe_events
-- -------------------------
-- Idempotency + audit/debug for webhook processing.
-- Store stripe_event_id to ensure you only process each event once.
CREATE TABLE IF NOT EXISTS stripe_events (
  stripe_event_id          text PRIMARY KEY,
  event_type               text NOT NULL,
  received_at              timestamptz NOT NULL DEFAULT now(),
  processed_at             timestamptz,
  processing_status        text NOT NULL DEFAULT 'received',
  error_message            text,
  related_customer_id      text,
  related_subscription_id  text,

  -- Optional: store payload. If you want lean DB, omit this and store only selected fields.
  payload                  jsonb,

  CONSTRAINT stripe_events_processing_status_allowed
    CHECK (processing_status IN ('received', 'processed', 'ignored', 'failed'))
);

CREATE INDEX IF NOT EXISTS idx_stripe_events_type_received
  ON stripe_events (event_type, received_at DESC);

CREATE INDEX IF NOT EXISTS idx_stripe_events_related_customer
  ON stripe_events (related_customer_id);

CREATE INDEX IF NOT EXISTS idx_stripe_events_related_subscription
  ON stripe_events (related_subscription_id);

-- -------------------------
-- 5) billing_sync_log (optional, but very useful)
-- -------------------------
-- Records the “ensure correct quantity” decisions and outcomes.
CREATE TABLE IF NOT EXISTS billing_sync_log (
  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id               uuid REFERENCES profiles(user_id) ON DELETE SET NULL,
  stripe_subscription_id text,
  stripe_event_id       text REFERENCES stripe_events(stripe_event_id) ON DELETE SET NULL,

  computed_years_out    integer NOT NULL,
  expected_quantity     integer NOT NULL,
  previous_quantity     integer,

  action_taken          text NOT NULL,
  created_at            timestamptz NOT NULL DEFAULT now(),

  CONSTRAINT billing_sync_log_years_out_min
    CHECK (computed_years_out >= 1),

  CONSTRAINT billing_sync_log_expected_quantity_min
    CHECK (expected_quantity >= 1),

  CONSTRAINT billing_sync_log_action_allowed
    CHECK (action_taken IN (
      'no_change',
      'updated_quantity',
      'missing_mapping',
      'skipped_not_active',
      'error'
    ))
);

CREATE INDEX IF NOT EXISTS idx_billing_sync_log_user_created
  ON billing_sync_log (user_id, created_at DESC);

CREATE INDEX IF NOT EXISTS idx_billing_sync_log_subscription_created
  ON billing_sync_log (stripe_subscription_id, created_at DESC);

-- -------------------------
-- 6) alumni_invites (optional, for alumni verification)
-- -------------------------
-- Useful if you want “only real alumni / only last N years can sign up”.
CREATE TABLE IF NOT EXISTS alumni_invites (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  email           text NOT NULL UNIQUE,
  graduation_year integer NOT NULL,
  invite_code     text NOT NULL UNIQUE,
  expires_at      timestamptz,
  used_at         timestamptz,
  created_at      timestamptz NOT NULL DEFAULT now(),

  CONSTRAINT alumni_invites_graduation_year_range
    CHECK (graduation_year BETWEEN 1900 AND 2100)
);

CREATE INDEX IF NOT EXISTS idx_alumni_invites_expires
  ON alumni_invites (expires_at);

CREATE INDEX IF NOT EXISTS idx_alumni_invites_used
  ON alumni_invites (used_at);
